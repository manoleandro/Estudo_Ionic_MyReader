                                                                                            import { InAppBrowser } from '@ionic-native/in-app-browser';

                                                                                            e altera o método itemSelected para:

                                                                                            itemSelected (url: string):void {
                                                                                            let browser = new InAppBrowser();
                                                                                            browser.create(url, '_system');
